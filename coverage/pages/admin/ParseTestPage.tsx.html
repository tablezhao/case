// 模拟拆分函数（与数据库逻辑保持一致）
const splitViolationText = (text: string): string[] => {
  if (!text || text.trim() === '') {
    return [];
  }
 
  // 按中文分号拆分
  const parts = text.split('；');
  
  // 清理空白字符并过滤空字符串
  return parts
    .map(part => part.trim())
    .filter(part => part.length > 0);
};
